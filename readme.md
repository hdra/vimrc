# Install
git clone https://github.com/hdra/vimrc ~/.vim

ln -s ~/.vim/vimrc ~/.vimrc

git submodule init

git submodule update

# Additional Binaries

## C-Tags

`brew install ctags`

## Go-Vim

run `:GoInstallBinaries` within vim

## Jsxhint

`npm install -g eslint`
`npm install -g babel-eslint`
`npm install -g eslint-plugin-react`
